66 std::getline(iss,tmp,',');

117inline Case2Data getCase2() {
118 Case2Data data2; // Criar estrutura para armazenar os dados
119
120 std::ifstream input2("../input/input1.txt");
121 if (!input2) {
122 std::cerr << "Error opening file!" << std::endl;
123 return data2; // Retorna estrutura vazia se não conseguir abrir o ficheiro
124 }
125
126 std::string line;
127 int n = 0;
128
129 while (std::getline(input2, line)) {
130 size_t pos = line.find(':');
131 if (pos == std::string::npos) continue; // Ignorar linhas inválidas
132
133 std::string value = line.substr(pos + 1);
134 while (!value.empty() && (value.front() == ' ' || value.front() == '\t')) {
135 value.erase(value.begin()); // Remover espaços no início
136 }
137
138 if (n == 0) {
139 data2.driving = (value == "driving");
140 }
141 else if (n == 1) {
142 data2.src = std::stoi(value);
143 }
144 else if (n == 2) {
145 data2.dest = std::stoi(value);
146 }
147 else if (n == 3) {
148 std::stringstream ss(value);
149 std::string token;
150 while (std::getline(ss, token, ',')) {
151 data2.avoid_nodes.push_back(std::stoi(token));
152 }
153 }
154 else if (n == 4) {
155 std::stringstream ss(value);
156 std::string token;
157 while (std::getline(ss, token, ')')) {
158 size_t start = token.find('(');
159 if (start != std::string::npos) {
160 std::string pair = token.substr(start + 1);
161 std::stringstream pairStream(pair);
162 std::string n1, n2;
163
164 if (std::getline(pairStream, n1, ',') && std::getline(pairStream, n2, ',')) {
165 data2.avoid_edges.emplace_back(std::stoi(n1), std::stoi(n2));
166 }
167 }
168 }
169 }
170 else if (n == 5) {
171 if (!value.empty()) {
172 data2.include_node = std::stoi(value);
173 } else {
174 data2.include_node = -1; // Definir um valor padrão caso esteja vazio
175 }
176 }
177 else {
178 break;
179 }
180
181 n++;
182 }
183
184 input2.close();
185 return data2;
186}
