82/*
83 * Auxiliary function to remove an outgoing edge (with a given destination (d))
84 * from a vertex (this).
85 * Returns true if successful, and false if such edge does not exist.
86 */
87template <class T>
88bool Vertex<T>::removeEdge(T in) {
89 bool removedEdge = false;
90 auto it = adj.begin();
91 while (it != adj.end()) {
92 Edge<T> *edge = *it;
93 Vertex<T> *dest = edge->getDest();
94 if (dest->getInfo() == in) {
95 it = adj.erase(it);
96 deleteEdge(edge);
97 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
98 }
99 else {
100 it++;
101 }
102 }
103 return removedEdge;
104}

113 Edge<T> *edge = *it;
114 it = adj.erase(it);
115 deleteEdge(edge);
116 }
117}

195template <class T>
196void Vertex<T>::deleteEdge(Edge<T> *edge) {
197 Vertex<T> *dest = edge->getDest();
198 // Remove the corresponding edge from the incoming list
199 auto it = dest->incoming.begin();
200 while (it != dest->incoming.end()) {
201 if ((*it)->getOrig()->getInfo() == info) {
202 it = dest->incoming.erase(it);
203 }
204 else {
205 it++;
206 }
207 }
208 delete edge;
209}

441/*
442 * Removes a vertex with a given content (in) from a graph (this), and
443 * all outgoing and incoming edges.
444 * Returns true if successful, and false if such vertex does not exist.
445 */
446template <class T>
447bool UrbanMap<T>::removeLocation(const T &in) {
448 for (auto it = locationSet.begin(); it != locationSet.end(); it++) {
449 if ((*it)->getInfo() == in) {
450 auto v = *it;
451 v->removeOutgoingEdges();
452 for (auto u : locationSet) {
453 u->removeEdge(v->getInfo());
454 }
455 locationSet.erase(it);
456 delete v;
457 return true;
458 }
459 }
460 return false;
461}

492template <class T>
493bool UrbanMap<T>::addBidirectionalRoad(const T &sourc, const T &dest, double driving, double walking) {
494 auto v1 = findLocation(sourc);
495 auto v2 = findLocation(dest);
496 if (v1 == nullptr || v2 == nullptr)
497 return false;
498 auto e1 = v1->addEdge(v2, driving, walking);
499 auto e2 = v2->addEdge(v1, driving, walking);
500 e1->setReverse(e2);
501 e2->setReverse(e1);
502 return true;
